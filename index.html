const CustomerFlow = () => {

  const [step, setStep] = useState(0);

  const [photo, setPhoto] = useState(null);

  const [prompt, setPrompt] = useState('');

  const [aiResults, setAiResults] = useState(null);

  const [regensLeft, setRegensLeft] = useState(3);

  const [generating, setGenerating] = useState(false);

  const [hasGenerated, setHasGenerated] = useState(false);

  const [selectedImage, setSelectedImage] = useState(null);

  const [selectedImageIdx, setSelectedImageIdx] = useState(null);

  const [variant, setVariant] = useState(null);

  const [completedOrder, setCompletedOrder] = useState(null);


  const resetAll = () => {

    setStep(0); setPhoto(null); setPrompt(''); setAiResults(null);

    setRegensLeft(3); setSelectedImage(null); setSelectedImageIdx(null);

    setVariant(null); setCompletedOrder(null); setHasGenerated(false);

  };


  const generateImages = async () => {

    setGenerating(true);

    const base64Image = photo.dataUrl.split(',')[1];


    try {

      const response = await fetch('/api/gemini', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({ prompt, imageBase64: base64Image, masterPrompt: MASTER_PROMPT })

      });

      const data = await response.json();

      

      if (data.images && data.images.length > 0) {

        setAiResults(data.images);

      } else {

        alert('Yapay zeka üretimi başarısız oldu. Lütfen tekrar deneyin.');

      }

    } catch (err) {

      console.error('API Hatası:', err);

      alert('Sunucu bağlantısı kurulamadı.');

    }


    if (hasGenerated) setRegensLeft(r => r - 1); else setHasGenerated(true);

    setGenerating(false);

  };


  const placeOrder = async (customerForm) => {

    const order = {

      id: `order_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`,

      status: 'pending',

      createdAt: new Date().toISOString(),

      customer: customerForm,

      photo: { dataUrl: photo.dataUrl, width: photo.width, height: photo.height, name: photo.name },

      prompt: prompt,

      masterPrompt: MASTER_PROMPT,

      aiResults: aiResults,

      selectedImageIndex: selectedImageIdx,

      selectedImage: selectedImage,

      variant: variant,

      regenerationsUsed: 3 - regensLeft,

    };

    await saveOrder(order);

    setCompletedOrder(order);

    setStep(6);

  };


  if (step === 0) return <Landing onStart={() => setStep(1)} />;

  if (step === 6 && completedOrder) return <SuccessPage order={completedOrder} onNew={resetAll} />;


  return (

    <div>

      <StepIndicator currentStep={step} />

      {step === 1 && <Step1Upload initialPhoto={photo} onNext={(p) => { setPhoto(p); setStep(2); }} />}

      {step === 2 && <Step2Prompt initialPrompt={prompt} onNext={(p) => { if (p !== prompt) { setAiResults(null); setHasGenerated(false); } setPrompt(p); setStep(3); }} onBack={() => setStep(1)} />}

      {step === 3 && <Step3Generate aiResults={aiResults} onGenerate={generateImages} generating={generating} regensLeft={regensLeft} onNext={(img, idx) => { setSelectedImage(img); setSelectedImageIdx(idx); setStep(4); }} onBack={() => setStep(2)} />}

      {step === 4 && <Step4Variant selectedVariant={variant} onNext={(v) => { setVariant(v); setStep(5); }} onBack={() => setStep(3)} />}

      {step === 5 && <Step5Checkout variant={variant} selectedImage={selectedImage} onNext={placeOrder} onBack={() => setStep(4)} />}

    </div>

  );

};

const AdminPanel = ({ onExit }) => {

  const [orders, setOrders] = useState([]);

  const [loading, setLoading] = useState(true);

  const [filter, setFilter] = useState('all');

  const [selectedOrder, setSelectedOrder] = useState(null);

  const [search, setSearch] = useState('');


  const refresh = async () => {

    setLoading(true);

    setOrders(await loadAllOrders());

    setLoading(false);

  };


  useEffect(() => { refresh(); }, []);


  const handleStatusUpdate = async (orderId, status) => {

    await updateOrderStatus(orderId, status);

    await refresh();

  };


  const filtered = orders.filter(o => {

    if (filter !== 'all' && o.status !== filter) return false;

    if (search) {

      const s = search.toLowerCase();

      return o.customer.firstName.toLowerCase().includes(s) ||

             o.customer.lastName.toLowerCase().includes(s) ||

             o.customer.email.toLowerCase().includes(s) ||

             shortId(o.id).toLowerCase().includes(s);

    }

    return true;

  });

    total: orders.length,

    pending: orders.filter(o => o.status === 'pending').length,

    approved: orders.filter(o => o.status === 'approved').length,

    shipped: orders.filter(o => o.status === 'shipped').length,

    revenue: orders.reduce((sum, o) => sum + (o.variant?.price || 0), 0),

            { label: 'Pending', value: stats.pending, color: 'text-yellow-600' },

            { label: 'Approved', value: stats.approved, color: 'text-green-600' },

            { label: 'Shipped', value: stats.shipped, color: 'text-blue-600' },

            </div>

          ))}

        </div>
